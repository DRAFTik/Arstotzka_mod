is_neutrality_ideology = {
	OR = {
		has_government = mysticist
		has_government = stratocratic_apologetism
		has_government = patrimonialist
	}
}

is_democratic_ideology = {
	OR = {
		has_government = market_liberal
		has_government = social_conservative
		has_government = stratocratic_apologetism
	}
}

is_fascism_ideology = {
	OR = {
		has_government = national_populist
		has_government = national_radicalist
	}
}

is_communism_ideology = {
	OR = {
		has_government = totalist
		has_government = syndicalist
		has_government = communitarianist
	}
}

is_neighbor_ideology = {
	OR = {
		AND = {
			is_fascism_ideology = yes
			FROM = { has_government = stratocratic_apologetism }
		}
		AND = {
			has_government = stratocratic_apologetism
			FROM = { is_fascism_ideology = yes }
		}
	}
}


is_enemy_ideology = {
	OR = {
		AND = {
			is_democratic_ideology = yes
			ROOT = {
				OR = {
					is_fascism_ideology = yes
					is_communism_ideology = yes
				}
			}
		}
		AND = {
			is_fascism_ideology = yes
			ROOT = {
				OR = {
					is_democratic_ideology = yes
					is_communism_ideology = yes
				}
			}
		}
		AND = {
			is_communism_ideology = yes
			ROOT = {
				OR = {
					is_democratic_ideology = yes
					is_fascism_ideology = yes
				}
			}
		}
	}
}
